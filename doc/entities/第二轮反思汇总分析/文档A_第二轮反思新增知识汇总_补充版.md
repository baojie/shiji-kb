```python
def identify_chapter_type(file_path):
    """章节类型识别"""
    with open(file_path) as f:
        title = f.readline()
        first_100_lines = f.readlines()[:100]

    # 从标题判断
    if '本纪' in title: base_type = '本纪'
    elif '世家' in title: base_type = '世家'
    elif '列传' in title: base_type = '列传'
    elif '表' in title: base_type = '表'
    elif '书' in title: base_type = '书'

    # 从内容判断特征
    content = ''.join(first_100_lines)
    if content.count('击') + content.count('攻') + content.count('破') > 20:
        feature = '军事'
    elif content.count('道') + content.count('德') + content.count('仁') > 15:
        feature = '哲学'
    # ... 其他特征判断

    return base_type, feature
```

**知识点5: 固定搭配模式自动化检查**

来源: 第二轮按章实体反思总结报告.md §5.1

核心内容:
```
固定搭配模式:

1. "以X为Y" → X人名, Y官职
   示例: 以〖@张良〗为〖;成信侯〗

2. "杀/诛/灭+实体" → 刑法动词标注
   示例: 〖[诛〗〖@孟说〗、〖[灭〗〖'赵〗

3. "作/著+书名" → 典籍标注
   示例: 〖[作〗〖%春秋〗、〖[著〗〖%道德经〗
```

价值:
- 可编写为正则表达式自动检查
- 高精度(95%+)低误报
- 适用于批量处理

自动化实现:
```python
import re

FIXED_PATTERNS = [
    {
        'name': '以X为Y',
        'pattern': r'以([^为]+)为([^。，；]+)',
        'rule': 'group(1)→人名, group(2)→官职',
        'check': lambda m: ('〖@' in m.group(1)) and ('〖;' in m.group(2))
    },
    {
        'name': '刑法+实体',
        'pattern': r'(杀|诛|灭|阬|弑)([^。，；〖]{1,4})',
        'rule': '动词→〖[刑法〗, 实体→对应类型',
        'check': lambda m: ('〖[' + m.group(1)) in text
    },
    {
        'name': '作/著+书名',
        'pattern': r'(作|著)([^。，；〖]{2,10})',
        'rule': '动词→〖[作〗, 书名→〖%典籍〗',
        'check': lambda m: ('〖%' in m.group(2))
    }
]

def check_fixed_patterns(text):
    """检查固定搭配模式"""
    errors = []
    for pattern_def in FIXED_PATTERNS:
        for match in re.finditer(pattern_def['pattern'], text):
            if not pattern_def['check'](match):
                errors.append({
                    'pattern': pattern_def['name'],
                    'text': match.group(0),
                    'rule': pattern_def['rule']
                })
    return errors
```

**知识点7: Python自动化检查脚本框架**

来源: 第二轮反思_核心规律提炼.md §4.2

核心内容:
```python
class EntityChecker:
    """实体标注一致性检查器"""

    def __init__(self, file_path):
        with open(file_path, 'r', encoding='utf-8') as f:
            self.content = f.read()
        self.errors = defaultdict(list)

    def check_punishment(self):
        """检查刑法词汇一致性"""
        punishments = ['诛', '杀', '灭', '阬', '弑', '烹', '醢', '殛']
        for word in punishments:
            pattern = f'[^〖]{word}[^〗]'
            matches = re.finditer(pattern, self.content)
            for match in matches:
                context = self.content[match.start()-20:match.start()+30]
                self.errors['刑法遗漏'].append((word, context))

    def check_identity(self):
        """检查身份词一致性"""
        identities = ['群臣', '百姓', '士卒', '诸侯', '父老', '子弟']
        for word in identities:
            pattern = f'{word}(?!〗)'
            if re.search(pattern, self.content):
                self.errors['身份遗漏'].append(word)

    def check_artifact(self):
        """检查器物词一致性"""
        artifacts = ['金', '衣', '车', '剑', '弓', '矢', '帛', '琴', '鼓']
        for word in artifacts:
            pattern = f'(?<!〖•){word}(?!〗)'
            matches = list(re.finditer(pattern, self.content))
            if matches:
                self.errors['器物遗漏'].append((word, len(matches)))

    def check_protagonist(self, name):
        """检查主人公省称"""
        full_pattern = f'〖@[^〗]*{name}[^〗]*〗'
        if re.search(full_pattern, self.content):
            short_pattern = f'(?<!〖@){name}(?!〗)'
            matches = list(re.finditer(short_pattern, self.content))
            if matches:
                self.errors['人名省称遗漏'].append((name, len(matches)))

    def check_battle_scene(self):
        """检查战争场景器物"""
        battle_keywords = ['战', '击', '攻', '破', '围', '拔', '败']
        weapon_keywords = ['剑', '弓', '矢', '戟', '戈', '矛', '盾', '甲', '车', '旗']

        for keyword in battle_keywords:
            for match in re.finditer(keyword, self.content):
                context = self.content[match.start():match.start()+200]
                for weapon in weapon_keywords:
                    if weapon in context and f'〖•{weapon}〗' not in context:
                        self.errors['战争场景器物遗漏'].append((weapon, context[:50]))
```

价值:
- 可复用的自动化检查框架
- 5大检查方法覆盖主要遗漏类型
- 可扩展,易于添加新检查规则

使用场景:
```python
# 使用示例
checker = EntityChecker('chapter_md/007_项羽本纪.tagged.md')
checker.check_punishment()
checker.check_identity()
checker.check_artifact()
checker.check_protagonist('羽')
checker.check_battle_scene()
checker.generate_report()
```
